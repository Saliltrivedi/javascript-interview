Flatten Nested Array with inbuilt method and without inbuilt + 
            
            function flatten(arr, res = []) {
              for (let i of arr) {
                Array.isArray(i) ? flatten(i, res) : res.push(i);
              }
              return res;
            }
            console.log(flatten([1,[2,[3,4]],5]));

            const arr = [1,[2,[3,4]],5];
            function flaten(arr, res = []) {
              for (let i = 0; i < arr.length; i++) {
                if (Array.isArray(arr[i])) {
                  flaten(arr[i], res);
                } else {
                  res.push(arr[i]);
                }
              }
              return res;
            }

        function areAnagrams(str1, str2) {
          if (str1.length !== str2.length) return false;

          const count = {};

          for (let char of str1) {
            count[char] = (count[char] || 0) + 1;
          }

          for (let char of str2) {
            if (!count[char]) return false;
            count[char]--;
          }

          return true;
        }

        console.log(areAnagrams("listen", "silent")); // true

Pure Function VS Impure Function with example+ 
              A pure function:
              Always returns the same output for the same input
              Does not modify external state (no side effects)

              Example:-
              function add(a, b) {
                return a + b;
              }

              add(2, 3); // 5
              add(2, 3); // 5 (always same result)

              An impure function:
              Uses or modifies external state
              Produces different outputs for the same input
              Has side effects


              let count = 0;

              function increment() {
                count++;
                return count;
              }
              increment(); // 1
              increment(); // 2 (different result)

the outer function has finished execution.
Without closure, we couldn’t preserve the running total across calls.”


Explain the stopping condition (important!)

“JavaScript cannot automatically know when to stop accepting arguments, so I explicitly stop the chain when the argument is undefined by calling the function with empty parentheses.”

  function flattenWithForLoop(current, result = {}) {
  const keys = Object.keys(current);

  for (let i = 0; i < keys.length; i++) {
    const key = keys[i];

    if (typeof current[key] === "object" && current[key] !== null) {
      flattenWithForLoop(current[key], result); // recursive call
    } else {
      result[key] = current[key]; // add primitive value to result
    }
  }

  return result;
}

                    function debounce(fn, delay) {
                      let timer;

                      return function (...args) {
                        clearTimeout(timer);
                        timer = setTimeout(() => {
                          fn.apply(this, args);
                        }, delay);
                      };
                    }
