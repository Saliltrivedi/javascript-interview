Program 47 . Implement your own map() / filter() / reduce(). 

        function myMap(arr, callback) {
        const result = [];
        for (let i = 0; i < arr.length; i++) {
            result.push(callback(arr[i], i, arr));
        }
        return result;
        }

        // Example:
        const nums = [1, 2, 3];
        const doubled = myMap(nums, (x) => x * 2);
        console.log(doubled); // [2, 4, 6]

        

        function myFilter(arr, callback) {
        const result = [];
        for (let i = 0; i < arr.length; i++) {
            if (callback(arr[i], i, arr)) {
            result.push(arr[i]);
            }
        }
        return result;
        }

        // Example:
        const nums = [1, 2, 3, 4];
        const evens = myFilter(nums, (x) => x % 2 === 0);
        console.log(evens); // [2, 4]

        function mergeStrings(s1, s2) {
            let maxLen = Math.max(s1.length, s2.length);
            let result = "";

            for (let i = 0; i < maxLen; i++) {
                if (s1[i]) result += s1[i];
                if (s2[i]) result += s2[i];
            }
            return result;
            }

            console.log(mergeStrings("abc", "123456"));   // a1b2c3456
            console.log(mergeStrings("abcdxyz", "123"));  // a1b2c3dxyz
